const data = Util.mock(6).circle();

  container: 'container',
  width: 500,
  height: 500,
  data,
  layout: {
    type: 'grid',
  },
  plugins: [
    {
      key: 'minimap',
      type: 'minimap',
      size: [300, 200],
      mode: 'delegate',
      delegateStyle: {
        fill: 'red',
      },
      className: 'g6-minimap-2',
      viewportClassName: 'g6-minimap-viewport-2',
    },
  ],
});
